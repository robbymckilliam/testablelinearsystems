{ %scope all the newcommands
  \newcommand{\plotyscale}{0.6}
  \newcommand{\arcdegree}{45}
  \makeatletter
  \pgfmathdeclarefunction{erf}{1}{ %I have no idea `which' error function this returns!
    \begingroup
    \pgfmathparse{#1 > 0 ? 1 : -1}
    \edef\sign{\pgfmathresult}%
    \pgfmathparse{abs(#1)}
    \edef\x{\pgfmathresult}%
    \pgfmathparse{1/(1+0.3275911*\x)}
    \edef\t{\pgfmathresult}%
    \pgfmathparse{%
      1 - (((((1.061405429*\t -1.453152027)*\t) + 1.421413741)*\t 
      -0.284496736)*\t + 0.254829592)*\t*exp(-(\x*\x))}
    \edef\y{\pgfmathresult}%
    \pgfmathparse{(\sign)*\y}
    \pgfmath@smuggleone\pgfmathresult
    \endgroup
  }

  \def\R{0.75}
  \def\B{1}
  \def\Kb{1}
  \def\Kt{1}
  \def\J{1}
  \def\thetafunc(#1){(3.14159/2*(1+erf(#1)))}
  \def\dthetafunc(#1){(sqrt(pi)*exp(-#1*#1))}
  \def\ddthetafunc(#1){(-2*#1*\dthetafunc(#1))}
  \def\vfunc(#1){(\thetafunc(#1) + (\R*\B/\Kt + \Kb)*\dthetafunc(#1) + \R\J/\Kt*\ddthetafunc(#1))}
  \newcommand{\background}[1]{

  \newcommand{\animations}[1]{
    \pgfmathparse{360/4*(1+erf(#1))}\pgfmathresult \let\motorangle\pgfmathresult
    % draw animated angular motor
    \begin{scope}[xshift=5.7cm,yshift=1.5cm]
      \node (C) [circle,minimum width=2.5cm] {};
      \draw[thick] (C.east)++(-0.075cm,0cm) -- +(0.15cm,0cm) node[pos=0.5,right] {0};
