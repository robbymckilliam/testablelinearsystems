%create pgf cross shape
\makeatletter
\pgfdeclareshape{cross}{%
  \inheritsavedanchors[from=circle]
  \inheritanchor[from=circle]{center}
  \inheritanchor[from=circle]{north}
  \inheritanchor[from=circle]{south}
  \inheritanchor[from=circle]{east}
  \inheritanchor[from=circle]{west}
  \backgroundpath{%
    % Save radius to x
    \pgf@x=\radius
    % Radius is also containing the "minimum width" and "minimum height"
    % This ensures that even with no text the shape will be drawn.
    % Unless of course that min are set to 0pt
    % So no need to check for that
    % Save radius
    \pgfutil@tempdima=\pgf@x%

    % draw cross
    \pgfpathmoveto{\pgfpoint{-1\pgf@x}{-1\pgf@x}}%
    \pgfpathlineto{{1\pgf@x}{1\pgf@x}}%
    \pgfpathclose
    \pgfpathmoveto{\pgfpoint{1\pgf@x}{-1\pgf@x}}%
    \pgfpathlineto{{1\pgf@x}{-1\pgf@x}}%
    \pgfpathclose

  }
}
